---
title: defaultLanguage
---

You can use the `defaultLanguage` string to get the default language in your application. For example:

```tsx title="app/[lang]/page.tsx"
// [!code highlight:1]
import { defaultLanguage } from '@/locales/.generated/types';
import { useRouter } from 'next/router';

export default function LanguageSelector() {
  const router = useRouter();
  return (
    <div>
      // [!code highlight:1]
      <button onClick={() => router.push(`/${defaultLanguage}`)}>
        {defaultLanguage}
      </button>
    </div>
  );
}
```
